[Visual Basic]

'
' VB Sample
'
Private Sub Print()
    Dim path As String
    path = "C:\Program Files\Brother bPAC3 SDK\Templates\NamePlate1.lbx"
    
    Dim doc As bpac.Document
    doc = CreateObject("bpac.Document")
    
    ' Open template
    doc.Open(path)
    
    ' Print setting start
    doc.StartPrint("", bpac.PrintOptionConstants.bpoDefault)
    
    ' Adds a print job (2 jobs are printed)
    doc.PrintOut(2, bpac.PrintOptionConstants.bpoDefault)
    
    ' Print setting end (=Print start)
    doc.EndPrint()
    
    ' Close template
    doc.Close()
End Sub

[Visual C++]
//
// C++ Sample

void Print() 
{
    CString strFilePath = _T("C:\\Program Files\\Brother bPAC3 SDK\\Templates\\NamePlate1.lbx");

    IDocumentPtr pDoc(__uuidof(bpac::Document));

    // Open template
    pDoc->Open(strFilePath.AllocSysString());

    // Print setting start
    pDoc->StartPrint(_bstr_t(_T("")), bpac::bpoDefault);

    // Adds a print job (2 jobs are printed)
    pDoc->PrintOut(2, bpac::bpoDefault);
    
    // Print setting end (=Print start)
    pDoc->EndPrint();

    // Close template
    pDoc->Close();
}
