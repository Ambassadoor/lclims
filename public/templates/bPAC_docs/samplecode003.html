[Visual C++]

// C++ Sample

const ULONG nCelt = 3;
ULONG nReturned = 0;
VARIANT arrVariant[nCelt] = {0};
IEnumVARIANTPtr enumPtr = doc->Objects->Get_NewEnum();
HRESULT hr = E_UNEXPECTED;
do {
    if (FAILED(hr = enumPtr->Next(nCelt, &arrVariant[0], &nReturned)))
        break;

    for (ULONG i = 0; i < nReturned; ++i)
    {
        bpac::IObjectPtr obj(arrVariant[i]);
        obj->Name;
    }
} while (hr != S_FALSE);
