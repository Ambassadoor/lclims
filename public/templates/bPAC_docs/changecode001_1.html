    ' Printer      :   Brother PT-9500PC
    ' TemplateFile :   C:\Program Files\Brother bPAC3 SDK\Templates\NamePlate1.lbx
    '
    Dim templateFile As String
    templateFile = "C:\Program Files\Brother bPAC3 SDK\Templates\NamePlate1.lbx"
    
    Dim objDoc As BrssCom.Document
    Set objDoc = CreateObject("BrssCom.Document")
    
    If (objDoc.Open(templateFile) <> False) Then
        objDoc.SetText 0, "Company"
        objDoc.SetText 1, "Name"
        objDoc.DoPrint bpoDefault, "0"
        objDoc.Close
    End If
    Set objDoc = Nothing

    Dim templateFile As String
    templateFile = "C:\Program Files\Brother bPAC3 SDK\Templates\NamePlate1.lbx"
    
    Dim objDoc As bpac.Document
    Set objDoc = CreateObject("bpac.Document")
    
    If (objDoc.Open(templateFile) <> False) Then
        objDoc.GetObject("objCompany").Text = "Company"
        objDoc.GetObject("objName").Text = "Name"
        objDoc.StartPrint "", bpoDefault
        objDoc.PrintOut 1, bpoDefault
        objDoc.EndPrint
        objDoc.Close
    End If
    Set objDoc = Nothing
