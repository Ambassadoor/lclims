    CString templateFile = _T("C:\\Program Files\\Brother bPAC3 SDK\\Templates\\NamePlate1.lbx");
    IDocumentPtr pObjDoc(__uuidof(bpac::Document));
    
    if (pObjDoc->Open(templateFile.AllocSysString()) != VARIANT_FALSE)
    {
        pObjDoc->GetObject(_bstr_t(_T("objCompany")))->PutText(_bstr_t(_T("Company")));
        pObjDoc->GetObject(_bstr_t(_T("objName")))->PutText(_bstr_t(_T("Name")));
        
        pObjDoc->StartPrint(_bstr_t(_T("")), bpac::bpoDefault);
        pObjDoc->PrintOut(1, bpac::bpoDefault);
        pObjDoc->EndPrint();
        pObjDoc->Close();
