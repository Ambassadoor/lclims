Private Sub Print()
    Dim path As String
    path = "C:\Program Files\Brother bPAC3 SDK\Templates\NamePlate1.lbx"
    Dim path2 As String
    path2 = "C:\Program Files\Brother bPAC3 SDK\Templates\NamePlate2.lbx"
    
    Dim doc As bpac.Document
    doc = CreateObject("bpac.Document")
    
    ' Print setting start
    doc.StartPrint("", bpac.PrintOptionConstants.bpoHalfCut)
    
    ' Open template
    doc.Open(path)
    
    ' Adds a print job
    doc.PrintOut(1, bpac.PrintOptionConstants.bpoDefault)
    
    ' Open template
    doc.Open(path2)
    
    ' Adds a print job
    doc.PrintOut(1, bpac.PrintOptionConstants.bpoDefault)
    
    ' Print setting end (=Print start)
    doc.EndPrint()
    
    ' Close template
    doc.Close()
End Sub
