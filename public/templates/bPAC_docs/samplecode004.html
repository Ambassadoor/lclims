Acquisition using object indexer 

[Visual Basic]

'
' VB Sample
'
Set objs = doc.Objects
For i = 0 To objs.Count - 1
    objs(i).Name
Next

[Visual C#]
//
// C# Sample
//
bpac.IObjects objs = doc.Objects;
for (int i = 0; i < objs.Count; ++i)
{
    objs[i].Name;
}

[Visual C++]
//
// C++ Sample
//
bpac::IObjectsPtr objs = doc->Objects;
for (int i = 0; i < objs->Count; ++i)
{
    objs->Item[i]->Name;
    //objs->GetItem(i)->Name;
}
