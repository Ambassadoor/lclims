[Visual Basic]

'
' VB Sample
'
Private Sub Print()
    Dim path As String
    path = "C:\Program Files\Brother bPAC3 SDK\Templates\NamePlate1.lbx"
    
    Dim strCompany(1) As String
    strCompany(0) = "ABC company"
    strCompany(1) = "XYZ corp."
    
    Dim strName(1) As String
    strName(0) = "Mike Smith"
    strName(1) = "John Brown"
    
    Dim doc As bpac.Document
    doc = CreateObject("bpac.Document")
    
    ' Open template
    doc.Open(path)
    
    ' Print setting start
    doc.StartPrint("", bpac.PrintOptionConstants.bpoHalfCut)
    
    For i As Integer = 0 To 1
         doc.GetObject("objCompany").Text = strCompany(i)
         doc.GetObject("objName").Text = strName(i)
        ' Adds a print job
         doc.PrintOut(1, bpac.PrintOptionConstants.bpoDefault)
    Next i
    
    ' Print setting end (=Print start)
    doc.EndPrint()
    
    ' Close template
    doc.Close()
End Sub

[Visual C#]

//
// C# Sample
//
private void Print()
{
    string path = "C:\\Program Files\\Brother bPAC3 SDK\\Templates\\NamePlate1.lbx";
    string[] strCompany = {"ABC company", "XYZ corp."};
    string[] strName = {"Mike Smith", "John Brown"};

    bpac.DocumentClass doc = new bpac.DocumentClass();

    // Open template
    doc.Open(path);

    // Print setting start
    doc.StartPrint("", bpac.PrintOptionConstants.bpoHalfCut);
    
    for (int i = 0; i < 2; i++) {
        doc.GetObject("objCompany").Text = strCompany[i];
        doc.GetObject("objName").Text = strName[i];
        // Adds a print job
        doc.PrintOut(1, bpac.PrintOptionConstants.bpoDefault);
    }

    // Print setting end (=Print start)
    doc.EndPrint();

    // Close template
    doc.Close();
}

//
// C++ Sample

void Print() 
{
    CString path = _T("C:\\Program Files\\Brother bPAC3 SDK\\Templates\\NamePlate1.lbx");
    CString strCompany[] = { _T("ABC company"), _T("XYZ corp.");
    CString strName[] = { _T("Mike Smith"), _T("John Brown");

    IDocumentPtr pObjDoc(__uuidof(Document));

    // Open template
    pObjDoc->Open(path.AllocSysString());

    // Print setting start
    pObjDoc->StartPrint(_bstr_t(_T("")), bpac::bpoDefault);

    for (int i = 0; i < 2; i++) {
        pObjDoc->GetObject(_bstr_t(_T("objCompany")))->PutText(strCompany[i].AllocSysString());
        pObjDoc->GetObject(_bstr_t(_T("objName")))->PutText(strName[i].AllocSysString());
        // Adds a print job
        pObjDoc->PrintOut(1, bpac::bpoDefault);
    }
    
    // Print setting end (=Print start)
    pObjDoc->EndPrint();

    // Close template
    pObjDoc->Close();
}
